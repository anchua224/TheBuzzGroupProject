import React, { useState,useEffect } from 'react';
import axios from 'axios';
import Comments from './components/Comments';
import { getSessionKey } from './Situation';
/**
 * Single idea part
 * @param {*} props variable from the idea 
 * @returns render of the single Idea
 * @module Idea
 */
export default function Idea(props) {

    /**
     * like count
     */
    const [like, setLike] = useState();

    /**
     * dislike accout
     */
    const [dislike, setDislike] = useState();

    /**
     * get current like count
     * and set the like
     */
    const getLike = async () => {
        axios.get(`https://cse216-fl22-team14.herokuapp.com/ideas/${props.id}/like`)
            .then(response =>{
                setLike(response.data.mData);
            })
            .catch(error => {
                console.log(error);
            })
    };

    /**
     * get current dislike count
     * and set the dislike
     */
    const getDislike = async () => {
        axios.get(`https://cse216-fl22-team14.herokuapp.com/ideas/${props.id}/dislike`)
            .then(response =>{
                setDislike(response.data.mData);
            })
            .catch(error => {
                console.log(error);
            })
    };

    /**
     * post a new like by user
     * get new value of like and dislike
     * @param {*} e the event of like 
     */
    const handleLike = async(e) =>{

        axios.post(`https://cse216-fl22-team14.herokuapp.com/ideas/${props.id}/like?sessionKey=${getSessionKey()}`)
            .catch(error => {
                console.log(error.massage)
            });

        getLike();
        getDislike();
    }

    /**
     * post a new dislike by user
     * get new value of like and dislike
     * @param {*} e the event of dislike 
     */
    const handleDisike = async(e) =>{


        axios.post(`https://cse216-fl22-team14.herokuapp.com/ideas/${props.id}/dislike?sessionKey=${getSessionKey()}`)
            .catch(error => {
                console.log(error.massage)
            });
            
        getLike();
        getDislike();
    }

    /**
     * usereffect: reload when like | dislike change
     */
    useEffect(()=>{
        getLike();
        getDislike();
    },[like,dislike]);

    /**
     * return html
     */
    return (
        <div className='idea' key={props.id}>
            <h3>{props.title}</h3>
            <h5>{props.massage}</h5>
            {/* <p>{props.createDate}</p> */}
            <button id={props.id}  onClick={(e) => handleLike(e)}>
                Like: {like}
            </button>
            <button id={props.id}  onClick={(e) => handleDisike(e)}>
                DisLike: {dislike}
            </button>
            <Comments 
                idea_id={props.id}
            />
        </div>
    )
}
